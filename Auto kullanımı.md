/*Burdaki olay auto'yu kullanarak integer,float,string yazmamıza
gerek kalmıyor otomatik bulup yazdırıyor bu baya pratik diyebilirim
*/

#include <iostream>
using namespace std;
int main()
{
auto x = 4; //integer
auto y = 3.37; //float
auto z = "merhaba galaksi"; //string
cout << x << endl;
cout << y << endl;
cout << z << endl;

Return 0;
}
